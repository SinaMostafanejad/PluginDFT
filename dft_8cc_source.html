164  Vb_ = std::shared_ptr<Matrix>(new Matrix(nso_,nso_));
165 
166  // orbital energies

168  epsilon_a_->copy(reference_wavefunction_->epsilon_a().get());

181

183  std::shared_ptr<Matrix> Sevec ( new Matrix(nso_,nso_) );
184  std::shared_ptr<Vector> Seval ( new Vector(nso_) );
185 
186  // build S^(-1/2) symmetric orthogonalization matrix
187  S_->diagonalize(Sevec,Seval);
188 
189  Shalf_ = (std::shared_ptr<Matrix>)( new Matrix(nso_,nso_) );
190  Shalf2 = (std::shared_ptr<Matrix>)( new Matrix(nso_,nso_) );
191  for (int mu = 0; mu < nso_; mu++) {
192  Shalf_->pointer()[mu][mu] = 1.0 / sqrt(Seval->pointer()[mu]);
193  Shalf2->pointer()[mu][mu] = sqrt(Seval->pointer()[mu]);
194  }
195 
196  // transform Seval back to nonorthogonal basis
197  Shalf_->back_transform(Sevec);
198  Shalf2->back_transform(Sevec);
199 
200 
201  // obtain phi(r), del phi(r)
202 
203  // evaluate basis function values on a grid:
204 
205  // the only way I can figure out to get a properly initialized grid is to
206  // read in a reference wave function from a previous dft job
207  scf::HF* scfwfn = (scf::HF*)reference_wavefunction_.get();
208  std::shared_ptr<SuperFunctional> functional = scfwfn->functional();
209  std::shared_ptr<VBase> potential = scfwfn->V_potential();
210 
211  // phi matrix (real-space <- AO basis mapping)
212  // since grid is stored in blocks, we need to build a full phi matrix
213  // from the individual blocks:
214  std::shared_ptr<PointFunctions> points_func = potential->properties()[0];
215  points_func->set_pointers(Da_,Db_);
216 
217  // determine number of grid points
218  int nblocks = potential->nblocks();
219  phi_points_ = 0;
220  int max_functions = 0;
221  int max_points = 0;
222  for (int myblock = 0; myblock < nblocks; myblock++) {
223  std::shared_ptr<BlockOPoints> block = potential->get_block(myblock);
224  points_func->compute_points(block);
225  int npoints = block->npoints();
226  phi_points_ += npoints;
227  const std::vector<int>& function_map = block->functions_local_to_global();
228  int nlocal = function_map.size();
229  if ( nlocal > max_functions ) max_functions = nlocal;
230  if ( npoints > max_points ) max_points = npoints;
231  }
232 
233  // what is the derivative level?
234  deriv_ = functional->deriv();
235  is_gga_ = functional->is_gga();
236  is_meta_ = functional->is_meta();
237  is_unpolarized_ = functional->is_unpolarized();
238 
239  super_phi_ = std::shared_ptr<Matrix>(new Matrix("SUPER PHI",phi_points_,nso_));
240 
241  if ( is_gga_ || is_meta_ ) {
242  super_phi_x_ = std::shared_ptr<Matrix>(new Matrix("SUPER PHI X",phi_points_,nso_));
243  super_phi_y_ = std::shared_ptr<Matrix>(new Matrix("SUPER PHI Y",phi_points_,nso_));
244  super_phi_z_ = std::shared_ptr<Matrix>(new Matrix("SUPER PHI Z",phi_points_,nso_));
245  }
246 
247  grid_x_ = std::shared_ptr<Vector>(new Vector("GRID X",phi_points_));
248  grid_y_ = std::shared_ptr<Vector>(new Vector("GRID Y",phi_points_));
249  grid_z_ = std::shared_ptr<Vector>(new Vector("GRID Z",phi_points_));
250  grid_w_ = std::shared_ptr<Vector>(new Vector("GRID W",phi_points_));
251 
252  // build phi matrix and derivative phi matrices
253  BuildPhiMatrix(potential, points_func, "PHI", super_phi_);
254 
255  if ( is_gga_ || is_meta_ ) {
256  BuildPhiMatrix(potential, points_func, "PHI_X",super_phi_x_);
257  BuildPhiMatrix(potential, points_func, "PHI_Y",super_phi_y_);
258  BuildPhiMatrix(potential, points_func, "PHI_Z",super_phi_z_);
259  }

262 void DFTSolver::BuildPhiMatrix(std::shared_ptr<VBase> potential, std::shared_ptr<PointFunctions> points_func,
263  std::string phi_type, std::shared_ptr<Matrix> myphi) {
264 
265  int nblocks = potential->nblocks();
266 
267  phi_points_ = 0;
268  for (int myblock = 0; myblock < nblocks; myblock++) {
269  std::shared_ptr<BlockOPoints> block = potential->get_block(myblock);
270  points_func->compute_points(block);
271  int npoints = block->npoints();
272  const std::vector<int>& function_map = block->functions_local_to_global();
273  int nlocal = function_map.size();
274 
275  double ** phi = points_func->basis_value(phi_type)->pointer();
276 
277  double * x = block->x();
278  double * y = block->y();
279  double * z = block->z();
280  double * w = block->w();
281 
282  for (int p = 0; p < npoints; p++) {
283 
284  grid_x_->pointer()[phi_points_ + p] = x[p];
285  grid_y_->pointer()[phi_points_ + p] = y[p];
286  grid_z_->pointer()[phi_points_ + p] = z[p];
287  grid_w_->pointer()[phi_points_ + p] = w[p];
288 
289  for (int nu = 0; nu < nlocal; nu++) {
290  int nug = function_map[nu];
291  myphi->pointer()[phi_points_ + p][nug] = phi[p][nu];
292  }
293  }
294  phi_points_ += npoints;
295  }
296  // grab AO->SO transformation matrix
297  std::shared_ptr<Matrix> ao2so = reference_wavefunction_->aotoso();
298 
299  // transform one index of super phi matrix (AO->SO)
300  std::shared_ptr<Matrix> temp (new Matrix(myphi));
301 
302  for (int p = 0; p < phi_points_; p++) {
303  for (int sigma = 0; sigma < nso_; sigma++) {
304  double dum = 0.0;
305  for (int nu = 0; nu < nso_; nu++) {
306  dum += myphi->pointer()[p][nu] * ao2so->pointer()[nu][sigma];
307  }
308  temp->pointer()[p][sigma] = dum;
309  }
310  }
311 
312  // AED: don't think we need this now. use this to do so-mo transformed phi later
313  C_DCOPY(nso_*phi_points_,temp->pointer()[0],1,myphi->pointer()[0],1);
314 /*
315  //
316  // p(r) = phi . D' . phi^T, with D' in the SO basis
317  //
318  // but, we're working with D in an orthonormal basis
319  //
320  // D = S^{1/2}.D'.S^{1/2}
321  //
322  // p(r) = (phi.S^{-1/2}).D.(phi.S^{-1/2})^T
323  //
324  // so, we need to tack on an additional S^{-1/2}
325  // to the phi matrix
326  //
327  for (int p = 0; p < phi_points_; p++) {
328  for (int sigma = 0; sigma < nso_; sigma++) {
329  double dum = 0.0;
330  for (int nu = 0; nu < nso_; nu++) {
331  dum += temp->pointer()[p][nu] * Shalf_->pointer()[nu][sigma];
332  }
333  super_phi_->pointer()[p][sigma] = dum;
334  }
335  }
336 */

339 double DFTSolver::compute_energy() {
340 
341  // grab the one-electron integrals from MintsHelper:
342  std::shared_ptr<MintsHelper> mints (new MintsHelper(reference_wavefunction_));
343 
344  // one-electron kinetic energy integrals
345  std::shared_ptr<Matrix> T = mints->so_kinetic();
346 
347  // one-electron potential energy integrals
348  std::shared_ptr<Matrix> V = mints->so_potential();
349 
350  // build the core hamiltonian
351  std::shared_ptr<Matrix> h = (std::shared_ptr<Matrix>)(new Matrix(T));
352  h->add(V);
353 
354  // get primary basis:
355  std::shared_ptr<BasisSet> primary = reference_wavefunction_->get_basisset("ORBITAL");
356 
357  // get auxiliary basis:
358  std::shared_ptr<BasisSet> auxiliary = reference_wavefunction_->get_basisset("DF_BASIS_SCF");
359 
360  // total number of auxiliary basis functions
361  int nQ = auxiliary->nbf();
362 
363  // determine the DFT functional and initialize the potential object
364  scf::HF* scfwfn = (scf::HF*)reference_wavefunction_.get();
365  std::shared_ptr<SuperFunctional> functional = scfwfn->functional();
366  std::shared_ptr<VBase> potential = VBase::build_V(primary,functional,options_,(options_.get_str("REFERENCE") == "RKS" ? "RV" : "UV"));
367 
368  potential->initialize();
369 
370  // print the ks information
371  potential->print_header();
372 
373  // JK object
374  std::shared_ptr<DiskDFJK> jk = (std::shared_ptr<DiskDFJK>)(new DiskDFJK(primary,auxiliary));
375 
376  // memory for jk (say, 80% of what is available)
377  jk->set_memory(0.8 * Process::environment.get_memory());
378 
379  // integral cutoff
380  jk->set_cutoff(options_.get_double("INTS_TOLERANCE"));
381 
382  // Do J/K/wK?
383  bool is_x_lrc = functional->is_x_lrc();
384  if ( options_["IP_FITTING"].has_changed() ) {
385  if ( options_.get_bool("IP_FITTING") ) {
386  is_x_lrc = true;
387  }
388  }
389  double x_omega = functional->x_omega();
390  if ( options_["DFT_OMEGA"].has_changed() ) {
391  x_omega = options_.get_double("DFT_OMEGA");
392  }
393 
394  jk->set_do_J(true);
395  jk->set_do_K(functional->is_x_hybrid());
396  jk->set_do_wK(is_x_lrc);
397  jk->set_omega(x_omega);
398 
399  jk->initialize();
400 
401  // grab some input options_
402  double e_convergence = options_.get_double("E_CONVERGENCE");
403  double d_convergence = options_.get_double("D_CONVERGENCE");
404  int maxiter = options_.get_int("MAXITER");
405 
406  outfile->Printf("\n");
407  outfile->Printf(" No. basis functions: %5i\n",nso_);
408  outfile->Printf(" No. auxiliary basis functions: %5i\n",nQ);
409  outfile->Printf(" No. alpha electrons: %5i\n",nalpha_);
410  outfile->Printf(" No. beta electrons: %5i\n",nbeta_);
411  outfile->Printf(" e_convergence: %10.3le\n",e_convergence);
412  outfile->Printf(" d_convergence: %10.3le\n",d_convergence);
413  outfile->Printf(" maxiter: %5i\n",maxiter);
414  outfile->Printf("\n");
415  outfile->Printf("\n");
416 
417  // form F' = ST^(-1/2) F S^(-1/2), where F = h
418  Fa_->copy(h);
419  Fb_->copy(h);
420 
421  std::shared_ptr<Matrix> Fprime_a ( new Matrix(Fa_) );
422  std::shared_ptr<Matrix> Fprime_b ( new Matrix(Fb_) );
423 
424  Fprime_a->transform(Shalf_);
425  Fprime_b->transform(Shalf_);
426 
427  // allocate memory for eigenvectors of F'
428  std::shared_ptr<Matrix> Fevec_a ( new Matrix(nso_,nso_) );
429 
430  std::shared_ptr<Matrix> Fevec_b ( new Matrix(nso_,nso_) );
431 
432  // diagonalize F' to obtain C'
433  Fprime_a->diagonalize(Fevec_a,epsilon_a_,ascending);
434  Fprime_b->diagonalize(Fevec_b,epsilon_b_,ascending);
435 
436  // Find C = S^(-1/2)C'
437  Ca_->gemm(false,false,1.0,Shalf_,Fevec_a,0.0);
438  Cb_->gemm(false,false,1.0,Shalf_,Fevec_b,0.0);
439 
440  // Construct density from C
441  C_DGEMM('n','t',nso_,nso_,nalpha_,1.0,&(Ca_->pointer()[0][0]),nso_,&(Ca_->pointer()[0][0]),nso_,0.0,&(Da_->pointer()[0][0]),nso_);
442  C_DGEMM('n','t',nso_,nso_,nbeta_,1.0,&(Cb_->pointer()[0][0]),nso_,&(Cb_->pointer()[0][0]),nso_,0.0,&(Db_->pointer()[0][0]),nso_);
443 
444  // initial energy, E = D(H+F) + Enuc
445  double e_current = enuc_;
446  e_current += 0.5 * Da_->vector_dot(h);
447  e_current += 0.5 * Db_->vector_dot(h);
448  e_current += 0.5 * Da_->vector_dot(Fa_);
449  e_current += 0.5 * Db_->vector_dot(Fb_);
450 
451  // SCF iterations
452 
453  double e_last = 0.0;
454  double dele = 0.0;
455  double gnorm_a = 0.0;
456  double gnorm_b = 0.0;
457 
458  outfile->Printf("\n");
459  outfile->Printf(" Guess energy: %20.12lf\n",e_current);
460  outfile->Printf("\n");
461  outfile->Printf(" ==> Begin SCF Iterations <==\n");
462  outfile->Printf("\n");
463  outfile->Printf(" ");
464  outfile->Printf(" Iter ");
465  outfile->Printf(" energy ");
466  outfile->Printf(" dE ");
467  outfile->Printf(" RMS |[F,P]| ");
468  outfile->Printf("\n");
469 
470  bool do_diis = options_.get_bool("DIIS");
471 
472  // Initialize our diis extrapolation manager. Note that,
473  // in this implemenation, the DIIS managar allocates
474  // memory for two buffers of the size of the Fock matrix.
475 
476  std::shared_ptr<DIIS> diis (new DIIS(nso_*nso_));
477 
478  int iter = 0;
479  do {
480 
481  e_last = e_current;
482 
483  // grab occupied alpha orbitals (the first na)
484  std::shared_ptr<Matrix> myCa (new Matrix(Ca_) );
485  myCa->zero();
486 
487  // grab occupied beta orbitals (the first nb)
488  std::shared_ptr<Matrix> myCb (new Matrix(Cb_) );
489  myCb->zero();
490 
491  for (int mu = 0; mu < nso_; mu++) {
492  for (int i = 0; i < nalpha_; i++) {
493  myCa->pointer()[mu][i] = Ca_->pointer()[mu][i];
494  }
495  for (int i = 0; i < nbeta_; i++) {
496  myCb->pointer()[mu][i] = Cb_->pointer()[mu][i];
497  }
498  }
499 
500  // push occupied orbitals onto JK object
501  std::vector< std::shared_ptr<Matrix> >& C_left = jk->C_left();
502  C_left.clear();
503  C_left.push_back(myCa);
504  C_left.push_back(myCb);
505 
506  // form J/K
507  jk->compute();
508 
509  // form Fa = h + Ja + Jb
510  Fa_->copy(h);
511  Fa_->add(jk->J()[0]);
512  Fa_->add(jk->J()[1]);
513 
514  // form Fb = h + Ja + Jb
515  Fb_->copy(h);
516  Fb_->add(jk->J()[0]);
517  Fb_->add(jk->J()[1]);
518 
519  // Construct density from C
520  C_DGEMM('n','t',nso_,nso_,nalpha_,1.0,&(Ca_->pointer()[0][0]),nso_,&(Ca_->pointer()[0][0]),nso_,0.0,&(Da_->pointer()[0][0]),nso_);
521  C_DGEMM('n','t',nso_,nso_,nbeta_, 1.0,&(Cb_->pointer()[0][0]),nso_,&(Cb_->pointer()[0][0]),nso_,0.0,&(Db_->pointer()[0][0]),nso_);
522 
523  if (functional->needs_xc()) {
524 
525  // set a/b densities in potential object
526  potential->set_D({Da_, Db_});
527 
528  // evaluate a/b potentials
529  potential->compute_V({Va_,Vb_});
530 
531  // form Fa/b = h + Ja + Jb + Va/b
532  Fa_->add(Va_);
533  Fb_->add(Vb_);
534 
535  }
536 
537  // exact exchange?
538  if (functional->is_x_hybrid()) {
539  // form F = h + 2*J + V - alpha K
540  double alpha = functional->x_alpha();
541  Fa_->axpy(-alpha,jk->K()[0]);
542  Fb_->axpy(-alpha,jk->K()[1]);
543  }
544 
545  // LRC functional?
546  if (is_x_lrc) {
547  // form Fa/b = h + Ja + Jb + Va/b - alpha Ka/b - beta wKa/b
548  double beta = 1.0 - functional->x_alpha();
549  Fa_->axpy(-beta,jk->wK()[0]);
550  Fb_->axpy(-beta,jk->wK()[1]);
551  }
552 
553  // evaluate the current energy
554  double one_electron_energy = Da_->vector_dot(h)
555  + Db_->vector_dot(h);
556 
557  double two_electron_energy = 0.5 * Da_->vector_dot(jk->J()[0])
558  + 0.5 * Da_->vector_dot(jk->J()[1])
559  + 0.5 * Db_->vector_dot(jk->J()[0])
560  + 0.5 * Db_->vector_dot(jk->J()[1]);
561  double HF_ex_energy = 0.0;
562  if (functional->is_x_hybrid()) {
563  double alpha = functional->x_alpha();
564  HF_ex_energy -= 0.5 * alpha * Da_->vector_dot(jk->K()[0]);
565  HF_ex_energy -= 0.5 * alpha * Db_->vector_dot(jk->K()[1]);
566  two_electron_energy += HF_ex_energy;
567  }
568 
569  double LR_Coulomb_energy = 0.0;
570  if (is_x_lrc) {
571  double beta = 1.0 - functional->x_alpha();
572  LR_Coulomb_energy -= 0.5 * beta * Da_->vector_dot(jk->wK()[0]);
573  LR_Coulomb_energy -= 0.5 * beta * Db_->vector_dot(jk->wK()[1]);
574  two_electron_energy += LR_Coulomb_energy;
575  }
576 
577  double exchange_correlation_energy = 0.0;
578  if (functional->needs_xc()) {
579  exchange_correlation_energy = potential->quadrature_values()["FUNCTIONAL"];
580 
581  //===================================
582  // Evaluate the mcpdft_xc_energy
583  //===================================
584 
585  BuildRho(Da_,Db_);
586 
587  if (options_.get_str("FUNCTIONAL") == "SVWN") {
588 
589  std::shared_ptr<Vector> rho = (std::shared_ptr<Vector>)(new Vector(phi_points_));
590 
591  std::shared_ptr<Vector> exc = (std::shared_ptr<Vector>)(new Vector(phi_points_));
592  rho->zero();
