375  }
376  }else if (deriv_ == 1) {
377 
378  for (int p = 0; p < phi_points_; p++) {

382  double rho = rho_ap[p] + rho_bp[p];

386  double rho2 = pow( rho, 2.0);
387

389  double g = pow(f,-2.0);
390  double F_PBE = 1.0 + KAPPA - KAPPA/f ;
391

398

400 
401  double sig = sigma_aap[p] + 2.0 * sigma_abp[p] + sigma_bbp[p];
402  double rho = rho_ap[p] + rho_bp[p];

404  double rho23 = pow( rho, 2.0/3.0);

407  double rho4 = pow( rho, 4.0);
408

411  double h = 1.0 / rho13 / rho2 * g;
412  double i = g / f;
413  double sig2 = pow( sigma_aap[p], 2.0);
414 
415  double F_PBE = 1.0 + KAPPA - KAPPA/f ;
416 
417  exc += -Cx * rho43 * F_PBE * grid_w_->pointer()[p];
418 
419  v2rho_a2p[p] = ( -0.6564966812284644 / rho23 * F_PBE ) - ( 0.02258606682377246 / rho13 / rho2 / rho * g * sig ) +
420  0.8590928633765426e-3 / rho4 / rho2 * i * sig2;
421  v2rho_a_sigma_aap[p] = 0.02258606682377246 * h - 0.644319647532407e-3 / rho4 / rho * i * sig;
422  v2sigma_aa2p[p] = 0.9664794712986105e-3 / rho4 * i;
423  }
424  }else {
425 
426  throw PsiException("The differentiation order should be 0, 1 or 2!",__FILE__,__LINE__);
427  }
