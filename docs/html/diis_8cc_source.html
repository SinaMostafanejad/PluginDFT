57 // Write current solution vector to disk (in file PSIF_DCC_OVEC).
58 void DIIS::WriteVector(double * vector1, double * vector2){
59 
60  // Name the entry in PSIF_DCC_OVEC according to the current
61  // DIIS iteration. If we already have maxdiis_ vectors, then
62  // replace one.
63  char * oldvector = (char*)malloc(1000*sizeof(char));
64  if ( diis_iter_ <= maxdiis_ ){
65  sprintf(oldvector,"oldvector%i",diis_iter_);
66  }
67  else{
68  sprintf(oldvector,"oldvector%i",replace_diis_iter_);
69  }
70 
71  std::shared_ptr<PSIO> psio(new PSIO());
72  if ( diis_iter_ == 0 ) {
73  psio->open(PSIF_DCC_OVEC,PSIO_OPEN_NEW);
74  }else {
75  psio->open(PSIF_DCC_OVEC,PSIO_OPEN_OLD);
76  }
77 
78  // Write the current solution vector.
79  psio_address addr = PSIO_ZERO;
80  psio->write(PSIF_DCC_OVEC,oldvector,(char*)vector1,dimdiis_*sizeof(double),addr,&addr);
81  psio->write(PSIF_DCC_OVEC,oldvector,(char*)vector2,dimdiis_*sizeof(double),addr,&addr);
82  psio->close(PSIF_DCC_OVEC,1);
83 
84  free(oldvector);
85 }

124 // Perform DIIS extrapolation.
125 void DIIS::Extrapolate(double * vector1, double * vector2){
126 
127  if ( diis_iter_ > 2 ) {
128 
129  // Compute coefficients for the extrapolation
130  DIISCoefficients( diis_iter_ < maxdiis_ ? diis_iter_ : maxdiis_ );
131 
132  memset((void*)vector1,'\0',dimdiis_*sizeof(double));
133  memset((void*)vector2,'\0',dimdiis_*sizeof(double));
134 
135  char * oldvector = (char*)malloc(1000*sizeof(char));
136 
137  std::shared_ptr<PSIO> psio(new PSIO());
138  psio->open(PSIF_DCC_OVEC,PSIO_OPEN_OLD);
139 
140  int max = diis_iter_;
141  if (max > maxdiis_) max = maxdiis_;
142 
143  // Read each of the old vectors from disk.
144  for (int j = 1; j <= max; j++){
145  sprintf(oldvector,"oldvector%i",j);
146  psio_address addr = PSIO_ZERO;
147  psio->read(PSIF_DCC_OVEC,oldvector,(char*)&tmp1_[0],dimdiis_*sizeof(double),addr,&addr);
148  // Accumulate extrapolated vector.
149  C_DAXPY(dimdiis_,diisvec_[j-1],tmp1_,1,vector1,1);
150 
151  psio->read(PSIF_DCC_OVEC,oldvector,(char*)&tmp1_[0],dimdiis_*sizeof(double),addr,&addr);
152  // Accumulate extrapolated vector.
153  C_DAXPY(dimdiis_,diisvec_[j-1],tmp1_,1,vector2,1);
154  }
155  psio->close(PSIF_DCC_OVEC,1);
156  free(oldvector);
157  }
158 
159  if (diis_iter_ <= maxdiis_){
160  diis_iter_++;
161  }
162  else {
163  // If we already have maxdiis_ vectors, choose the one with
164  // the largest error as the one to replace.
165  std::shared_ptr<PSIO> psio(new PSIO());
166  psio->open(PSIF_DCC_EVEC,PSIO_OPEN_OLD);
167  int jmax = 1;
168  double max = -1.0e99;
169  char * evector = (char*)malloc(1000*sizeof(char));
170  for (int j = 1; j <= maxdiis_; j++){
171  sprintf(evector,"evector%i",j);
172  psio_address addr = PSIO_ZERO;
173  psio->read(PSIF_DCC_EVEC,evector,(char*)tmp1_,dimdiis_*sizeof(double),addr,&addr);
174  double nrm = C_DNRM2(dimdiis_,tmp1_,1);
175  psio->read(PSIF_DCC_EVEC,evector,(char*)tmp1_,dimdiis_*sizeof(double),addr,&addr);
176  nrm += C_DNRM2(dimdiis_,tmp1_,1);
177  nrm = 0.5 * nrm;
178  if ( nrm > max ) {
179  max = nrm;
180  jmax = j;
181  }
182  }
183  psio->close(PSIF_DCC_EVEC,1);
184  replace_diis_iter_ = jmax;
185  free(evector);
186  }
187  //else if (replace_diis_iter_ < maxdiis_) replace_diis_iter_++;
188  //else replace_diis_iter_ = 1;
189 }

191 // Evaluate extrapolation coefficients for DIIS.
192 void DIIS::DIISCoefficients(int nvec){
193 
194  // Allocate memory for small matrices/vectors.
195  int * ipiv = (int*)malloc((nvec+1)*sizeof(int));
196  double * temp = (double*)malloc(sizeof(double)*maxdiis_*maxdiis_);
197  double * A = (double*)malloc(sizeof(double)*(nvec+1)*(nvec+1));
198  double * B = (double*)malloc(sizeof(double)*(nvec+1));
199  memset((void*)A,'\0',(nvec+1)*(nvec+1)*sizeof(double));
200  memset((void*)B,'\0',(nvec+1)*sizeof(double));
201  B[nvec] = -1.;
202 
203  char * evector = (char*)malloc(1000*sizeof(char));
204 
205  std::shared_ptr<PSIO> psio(new PSIO());
206  psio->open(PSIF_DCC_EVEC,PSIO_OPEN_OLD);
207 
208  // Read in the previous error matrix, so we don't have
209  // to build the entire thing each iteration.
210  psio->read_entry(PSIF_DCC_EVEC,"error matrix",(char*)&temp[0],maxdiis_*maxdiis_*sizeof(double));
211 
212  // Reshape the error matrix, in case its dimension is less than maxdiis_.
213  for (int i = 0; i < nvec; i++){
214  for (int j = 0; j < nvec; j++){
215  A[i*(nvec+1)+j] = temp[i*maxdiis_+j];
216  }
217  }
218 
219  if (nvec <= 3) {
220  // At early iterations, just build the whole matrix.
221 
222 
223  for (int i = 0; i < nvec; i++) {
224  sprintf(evector,"evector%i",i+1);
225  psio->read_entry(PSIF_DCC_EVEC,evector,(char*)tmp1_,2*dimdiis_*sizeof(double));
226  for (int j = i+1; j < nvec; j++){
227  sprintf(evector,"evector%i",j+1);
228  psio->read_entry(PSIF_DCC_EVEC,evector,(char*)tmp2_,2*dimdiis_*sizeof(double));
229  double sum = C_DDOT(2*dimdiis_,tmp1_,1,tmp2_,1);
230  A[i*(nvec+1)+j] = sum;
231  A[j*(nvec+1)+i] = sum;
232  }
233  double sum = C_DDOT(2*dimdiis_,tmp1_,1,tmp1_,1);
234  A[i*(nvec+1)+i] = sum;
235  }
236 
237  }else {
238  // At later iterations, don't build the whote matrix.
239  // Just replace one row/column.
240 
241  // Which row/column will be replaced?
242  int i = nvec < maxdiis_ ? nvec - 1 : replace_diis_iter_ - 1;
243 
244  sprintf(evector,"evector%i",i+1);
245  psio->read_entry(PSIF_DCC_EVEC,evector,(char*)tmp1_,2*dimdiis_*sizeof(double));
246  for (int j = 0; j < nvec; j++){
247  sprintf(evector,"evector%i",j+1);
248  psio->read_entry(PSIF_DCC_EVEC,evector,(char*)tmp2_,2*dimdiis_*sizeof(double));
249  double sum = C_DDOT(2*dimdiis_,tmp1_,1,tmp2_,1);
250  A[i*(nvec+1)+j] = sum;
251  A[j*(nvec+1)+i] = sum;
252  }
253  }
254 
255  int j = nvec;
256  for (int i = 0; i < (nvec+1); i++){
257  A[j*(nvec+1)+i] = -1.0;
258  A[i*(nvec+1)+j] = -1.0;
259  }
260  A[(nvec+1)*(nvec+1)-1] = 0.0;
261 
262  // Write error matrix for next iteration.
263  for (int i = 0; i < nvec; i++){
264  for (int j = 0; j < nvec; j++){
265  temp[i*maxdiis_+j] = A[i*(nvec+1)+j];
266  }
267  }
268  psio->write_entry(PSIF_DCC_EVEC,"error matrix",(char*)&temp[0],maxdiis_*maxdiis_*sizeof(double));
269  free(temp);
270  psio->close(PSIF_DCC_EVEC,1);
271  free(evector);
272 
273  // Solve the set of linear equations for the extrapolation coefficients
274  int nrhs = 1;
275  int lda = nvec+1;
276  int ldb = nvec+1;
277  C_DGESV(nvec+1,nrhs,A,lda,ipiv,B,ldb);
278  C_DCOPY(nvec,B,1,diisvec_,1);
279 
280  //outfile->Printf("\n");
281  //outfile->Printf(" ==> DIIS Expansion Coefficients <==\n");
282  //outfile->Printf("\n");
283  //for (int i = 0; i < nvec; i++) outfile->Printf(" c[%i] %20.12lf\n",i,diisvec_[i]);
284  //outfile->Printf("\n");
285 
286  free(A);
287  free(B);
288  free(ipiv);
289 }
